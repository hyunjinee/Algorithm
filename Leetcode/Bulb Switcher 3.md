var numTimesAllBlue = function (light) {
  let onCount = 0,
    furthestRight = -1,
    result = 0;

  for (let currentBulb of light) {
    onCount++;
    furthestRight = Math.max(furthestRight, currentBulb);
    if (onCount === furthestRight) result++;
  }

  return result;
};
