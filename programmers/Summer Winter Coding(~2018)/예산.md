def solution(d, budget):
    d.sort()
    while budget < sum(d):
        d.pop()
    return len(d)

function solution(d, budget) {
    d.sort((a, b) => a - b);

    while (d.reduce((a, b) => (a + b), 0) > budget) d.pop();

    return d.length;
