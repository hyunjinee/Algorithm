import sys

N, S = map(int, sys.stdin.readline().split(" "))
arr = list(map(int, sys.stdin.readline().split(" ")))

start = 0
end = 1
count = 0
sum = arr[start]
if sum == S:
    count += 1

while not (start == end == N):
    if sum < S and end < N:
        sum += arr[end]
        end += 1
    else:
        sum -= arr[start]
        start += 1

    if sum == S:
        count += 1


print(count)
